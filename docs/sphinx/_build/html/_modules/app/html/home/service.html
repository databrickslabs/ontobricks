def get_ontology_status(project) -> Dict[str, Any]:
    """Get ontology status for validation indicators.
    
    Args:
        project: ProjectSession instance
        
    Returns:
        dict: Ontology status with validation info
    """
    classes = project.get_classes()
    properties = project.get_properties()
    
    has_ontology = len(classes) > 0
    is_valid = has_ontology and all(
        c.get('uri') or c.get('name') or c.get('localName') 
        for c in classes
    )
    
    return {
        'success': True,
        'has_ontology': has_ontology,
        'is_valid': is_valid,
        'class_count': len(classes),
        'property_count': len(properties)
    }




def _active_classes_and_properties(project):
    """Return (active_classes, active_object_properties, excluded_names) excluding entities/relationships marked as excluded.
    
    The ``excluded`` flag lives on each entry in ``entities``
    and ``relationships``.
    """
    all_classes = project.get_classes()
    all_properties = project.get_properties()
    assignment = project.assignment or {}
    
    excluded_entity_uris = {
        m.get('ontology_class') for m in assignment.get('entities', [])
        if m.get('excluded')
    }
    excluded_rel_uris = {
        m.get('property') for m in assignment.get('relationships', [])
        if m.get('excluded')
    }
    
    active_classes = [c for c in all_classes if c.get('uri') not in excluded_entity_uris]
    excluded_names = set()
    for c in all_classes:
        if c.get('uri') in excluded_entity_uris:
            if c.get('name'):
                excluded_names.add(c['name'])
            if c.get('localName'):
                excluded_names.add(c['localName'])
    
    object_properties = [p for p in all_properties if p.get('type') == 'ObjectProperty']
    active_props = [
        p for p in object_properties
        if p.get('uri') not in excluded_rel_uris
        and p.get('domain') not in excluded_names
        and p.get('range') not in excluded_names
    ]
    return active_classes, active_props, excluded_names

def validate_status(project) -> Dict[str, Any]:
    """Get validation status for navbar indicators.

    Results are cached on the session and only recomputed when
    ``project.save()`` detects an ontology or assignment change.

    Args:
        project: ProjectSession instance
        
    Returns:
        dict: Complete validation status
    """
    cached = project.validation_cache
    if cached is not None:
        cached['ontology_changed'] = project.ontology_changed
        cached['assignment_changed'] = project.assignment_changed
        return cached

    logger.debug("Validation cache miss — recomputing")
    result = _compute_validation(project)
    project.validation_cache = result
    return result




def _compute_validation(project) -> Dict[str, Any]:
    """Run the full ontology + mapping validation rules."""
    all_classes = project.get_classes()
    all_properties = project.get_properties()
    active_classes, active_props, _ = _active_classes_and_properties(project)
    entity_mappings = project.get_entity_mappings()
    relationship_mappings = project.get_relationship_mappings()
    
    # Ontology validation (uses all classes – exclusion doesn't change ontology validity)
    ontology_issues = []
    for cls in all_classes:
        if not cls.get('uri') and not cls.get('name') and not cls.get('localName'):
            ontology_issues.append(f"Entity '{cls.get('label', 'Unknown')}' has no URI")
    
    ontology_valid = len(all_classes) > 0 and len(ontology_issues) == 0
    
    if len(all_classes) == 0:
        ontology_issues.append('No entities defined')
    
    # Mapping validation – only count active (non-excluded) items
    mapping_by_class = {m.get('ontology_class'): m for m in entity_mappings}
    mapped_class_uris = set(mapping_by_class.keys())
    mapped_property_uris = {m.get('property') for m in relationship_mappings}
    
    unmapped_entities_count = sum(
        1 for cls in active_classes 
        if (cls.get('uri') or cls.get('name')) not in mapped_class_uris
    )
    unmapped_relationships_count = sum(
        1 for prop in active_props 
        if (prop.get('uri') or prop.get('name')) not in mapped_property_uris
    )
    
    # Check attribute assignments for each mapped active entity
    unmapped_attributes_count = 0
    for cls in active_classes:
        cls_uri = cls.get('uri') or cls.get('name')
        data_props = cls.get('dataProperties', [])
        if not data_props or cls_uri not in mapped_class_uris:
            continue
        em = mapping_by_class.get(cls_uri, {})
        attr_map = em.get('attribute_mappings', {})
        for dp in data_props:
            attr_name = dp.get('name') or dp.get('localName') or ''
            if attr_name and attr_name not in attr_map:
                unmapped_attributes_count += 1
    
    mapping_issues = []
    if len(entity_mappings) == 0 and len(active_classes) > 0:
        mapping_issues.append('No entity mappings defined')
    elif unmapped_entities_count > 0:
        mapping_issues.append(f'{unmapped_entities_count} entity(ies) not mapped')
    
    if len(active_props) > 0:
        if len(relationship_mappings) == 0:
            mapping_issues.append('No relationship mappings defined')
        elif unmapped_relationships_count > 0:
            mapping_issues.append(f'{unmapped_relationships_count} relationship(s) not mapped')
    
    if unmapped_attributes_count > 0:
        mapping_issues.append(f'{unmapped_attributes_count} attribute(s) not assigned')
    
    mapping_valid = (
        len(active_classes) > 0 and 
        unmapped_entities_count == 0 and 
        unmapped_relationships_count == 0 and
        unmapped_attributes_count == 0
    )
    
    return {
        'ontology_valid': ontology_valid,
        'ontology_issues': ontology_issues,
        'ontology_stats': {
            'classes': len(all_classes),
            'properties': len(all_properties)
        },
        'ontology_changed': project.ontology_changed,
        'mapping_valid': mapping_valid,
        'mapping_issues': mapping_issues,
        'mapping_stats': {
            'entities': len(entity_mappings),
            'relationships': len(relationship_mappings),
            'total_classes': len(active_classes),
            'total_properties': len(active_props)
        },
        'assignment_changed': project.assignment_changed,
    }

def validate_ontology(project) -> Dict[str, Any]:
    """Validate current ontology.
    
    Args:
        project: ProjectSession instance
        
    Returns:
        dict: Validation result with errors
    """
    classes = project.get_classes()
    properties = project.get_properties()
    
    if not classes:
        return {
            'valid': False, 
            'message': 'No ontology loaded', 
            'errors': ['No classes defined']
        }
    
    errors = []
    for cls in classes:
        # Lenient: accept uri, name, or localName
        if not cls.get('uri') and not cls.get('name') and not cls.get('localName'):
            label = cls.get('label', cls.get('name', 'Unknown'))
            errors.append(f"Entity '{label}' has no URI")
    
    for prop in properties:
        if not prop.get('uri') and not prop.get('name') and not prop.get('localName'):
            label = prop.get('label', prop.get('name', 'Unknown'))
            errors.append(f"Relationship '{label}' has no URI")
    
    return {
        'valid': len(errors) == 0,
        'message': 'Valid' if not errors else f'{len(errors)} issues found',
        'errors': errors
    }

def validate_mapping(project) -> Dict[str, Any]:
    """Validate current mapping.
    
    Args:
        project: ProjectSession instance
        
    Returns:
        dict: Mapping validation with coverage info
    """
    entity_mappings = project.get_entity_mappings()
    active_classes, _, _ = _active_classes_and_properties(project)
    
    if not entity_mappings:
        return {
            'valid': False, 
            'complete': False, 
            'message': 'No entity mappings', 
            'coverage': 0
        }
    
    mapped_class_uris = {m.get('ontology_class') for m in entity_mappings}
    class_uris = {c.get('uri') or c.get('name') for c in active_classes}
    
    coverage = len(mapped_class_uris & class_uris) / len(class_uris) * 100 if class_uris else 0
    
    return {
        'valid': len(entity_mappings) > 0,
        'complete': coverage >= 100,
        'message': f'{int(coverage)}% coverage',
        'coverage': round(coverage, 1)
    }

def get_detailed_validation(project, warehouse_id: str = "") -> Dict[str, Any]:
    """Get detailed validation status.
    
    Args:
        project: ProjectSession instance
        warehouse_id: resolved warehouse ID (empty string if none)
        
    Returns:
        dict: Comprehensive validation details
    """
    all_classes = project.get_classes()
    all_properties = project.get_properties()
    active_classes, active_props, _ = _active_classes_and_properties(project)
    entity_mappings = project.get_entity_mappings()
    relationship_mappings = project.get_relationship_mappings()
    design_layout = project.design_layout
    
    # Ontology validation (uses all classes)
    ontology_errors = []
    for cls in all_classes:
        if not cls.get('uri') and not cls.get('name') and not cls.get('localName'):
            ontology_errors.append(f"Entity '{cls.get('label', 'Unknown')}' has no URI")
    
    # Mapping validation – only active items
    mapping_by_class = {m.get('ontology_class'): m for m in entity_mappings}
    mapped_class_uris = set(mapping_by_class.keys())
    mapped_property_uris = {m.get('property') for m in relationship_mappings}
    
    unmapped_entities = []
    for cls in active_classes:
        uri = cls.get('uri') or cls.get('name')
        if uri not in mapped_class_uris:
            unmapped_entities.append({
                'name': cls.get('name', ''),
                'label': cls.get('label', cls.get('name', 'Unknown')),
                'uri': cls.get('uri', '')
            })
    
    unmapped_relationships = []
    for prop in active_props:
        uri = prop.get('uri') or prop.get('name')
        if uri not in mapped_property_uris:
            unmapped_relationships.append({
                'name': prop.get('name', ''),
                'label': prop.get('label', prop.get('name', 'Unknown')),
                'uri': prop.get('uri', ''),
                'domain': prop.get('domain', ''),
                'range': prop.get('range', '')
            })
    
    # Check attribute assignments for each mapped active entity
    unmapped_attributes = []
    for cls in active_classes:
        cls_uri = cls.get('uri') or cls.get('name')
        data_props = cls.get('dataProperties', [])
        if not data_props or cls_uri not in mapped_class_uris:
            continue
        em = mapping_by_class.get(cls_uri, {})
        attr_map = em.get('attribute_mappings', {})
        cls_label = cls.get('label') or cls.get('name', 'Unknown')
        for dp in data_props:
            attr_name = dp.get('name') or dp.get('localName') or ''
            if attr_name and attr_name not in attr_map:
                unmapped_attributes.append({
                    'class': cls_label,
                    'attribute': attr_name
                })
    
    # Design validation
    has_design = bool(design_layout and design_layout.get('views'))
    
    # Overall status
    ontology_valid = len(all_classes) > 0 and len(ontology_errors) == 0
    
    # Mapping issues and warnings
    mapping_issues = []
    mapping_warnings = []
    
    if len(entity_mappings) == 0 and len(active_classes) > 0:
        mapping_issues.append('No entity mappings defined')
    elif len(unmapped_entities) > 0:
        mapping_issues.append(f'{len(unmapped_entities)} entity(ies) not mapped')
    
    if len(active_props) > 0:
        if len(relationship_mappings) == 0:
            mapping_issues.append('No relationship mappings defined')
        elif len(unmapped_relationships) > 0:
            mapping_issues.append(f'{len(unmapped_relationships)} relationship(s) not mapped')
    
    if len(unmapped_attributes) > 0:
        mapping_issues.append(f'{len(unmapped_attributes)} attribute(s) not assigned')
    
    # Mapping is complete ONLY if ALL active classes, ObjectProperties, and attributes are mapped
    all_classes_mapped = len(unmapped_entities) == 0 and len(active_classes) > 0
    all_properties_mapped = len(unmapped_relationships) == 0
    all_attributes_mapped = len(unmapped_attributes) == 0
    mapping_complete = all_classes_mapped and all_properties_mapped and all_attributes_mapped
    mapping_valid = mapping_complete and len(mapping_issues) == 0
    
    # Ontology warnings
    ontology_warnings = []
    ontology_name = project.ontology.get('name', '')
    if not ontology_name or ontology_name == 'MyOntology':
        ontology_warnings.append('Ontology name is still set to default')
    if len(active_props) == 0:
        ontology_warnings.append('No relationships defined')
    props_without_domain = [p for p in active_props if not p.get('domain')]
    if props_without_domain:
        ontology_warnings.append(f'{len(props_without_domain)} relationship(s) without domain')
    
    dtwin = _compute_dtwin_indicator(project)

    # Pull cached DT existence & triplestore status from the session so the
    # Project Status page can show the same detail as the DT Information page
    # without issuing extra I/O calls.
    ts_cache = (project.triplestore or {}).get("stats", {})
    dt_existence = ts_cache.get("dt_existence") or {}
    ts_status = ts_cache.get("status") or {}

    last_update = project.last_update
    last_build = project.last_build
    needs_rebuild = (last_update > last_build) if (last_update and last_build) else bool(last_update and not last_build)

    dtwin_detail = {
        **dtwin,
        'view_exists': dt_existence.get('view_exists'),
        'view_table': dt_existence.get('view_table', ''),
        'graph_name': dt_existence.get('graph_name', ''),
        'local_lbug_exists': dt_existence.get('local_lbug_exists', False),
        'registry_lbug_exists': dt_existence.get('registry_lbug_exists'),
        'snapshot_table': dt_existence.get('snapshot_table', ''),
        'snapshot_exists': dt_existence.get('snapshot_exists'),
        'last_built': last_build,
        'last_update': last_update,
        'has_data': ts_status.get('has_data', False),
        'triple_count': ts_status.get('count', 0),
        'needs_rebuild': needs_rebuild,
    }

    # Metadata table count (in-session, no I/O)
    metadata = project._data.get('metadata', {})
    metadata_table_count = len(metadata.get('tables', [])) if metadata else 0

    # Document count from session cache (set by document list/upload operations)
    document_count = project._data.get('_document_count')

    project_info = project.info or {}
    project_name = project_info.get('name', 'NewProject')
    project_version = project_info.get('version', '1')

    return {
        'success': True,
        # Flat format expected by the JavaScript
        'ontology_valid': ontology_valid,
        'mapping_valid': mapping_valid,
        'mapping_complete': mapping_complete,
        'ontology_stats': {
            'classes': len(all_classes),
            'properties': len(active_props)
        },
        'mapping_stats': {
            'entities': len(entity_mappings),
            'relationships': len(relationship_mappings),
            'total_classes': len(active_classes),
            'total_properties': len(active_props)
        },
        'ontology_issues': ontology_errors,
        'ontology_warnings': ontology_warnings,
        'mapping_issues': mapping_issues,
        'mapping_warnings': mapping_warnings,
        'unmapped_entities': unmapped_entities,
        'unmapped_relationships': unmapped_relationships,
        'unmapped_attributes': unmapped_attributes,
        # Digital Twin status (summary + detail)
        'dtwin': dtwin_detail,
        # Warehouse status
        'warehouse': {'warehouse_id': warehouse_id},
        # Project assets
        'metadata_table_count': metadata_table_count,
        'document_count': document_count,
        # Project identity
        'project_name': project_name,
        'project_version': project_version,
        # Also include nested format for other consumers
        'ontology': {
            'valid': ontology_valid,
            'entity_count': len(all_classes),
            'relationship_count': len(active_props),
            'errors': ontology_errors
        },
        'mapping': {
            'valid': mapping_valid,
            'complete': mapping_complete,
            'entity_count': len(entity_mappings),
            'relationship_count': len(relationship_mappings)
        },
        'design': {
            'valid': has_design,
            'has_layout': has_design
        },
        'ready_for_query': ontology_valid and mapping_complete
    }




def _compute_dtwin_indicator(project) -> Dict[str, Any]:
    """Derive a three-state Digital Twin indicator from cached data.

    Returns a dict with:
        indicator: 'green' | 'orange' | 'red'
        title:     tooltip text for the navbar
        count:     triple count (0 when unknown)
        pending:   True when no status has been fetched yet
    """
    stats = (project.triplestore or {}).get("stats", {})
    ts_status = stats.get("status")
    dt_exist = stats.get("dt_existence")

    if not ts_status and not dt_exist:
        if not project.last_build:
            logger.debug("DT indicator: never built (red)")
            return {"indicator": "red", "title": "Digital Twin never built",
                    "count": 0, "pending": False}
        logger.debug("DT indicator: status not yet checked (orange)")
        return {"indicator": "orange", "title": "Digital Twin status not yet checked",
                "count": 0, "pending": True}

    graph_loaded = bool(
        ts_status and ts_status.get("has_data") and ts_status.get("count", 0) > 0
    )
    count = (ts_status or {}).get("count", 0)

    view_exists = (dt_exist or {}).get("view_exists")
    archive_exists = (dt_exist or {}).get("registry_lbug_exists")

    # Green: graph is loaded AND view either confirmed or not yet checked
    # (a successful build creates both, so graph loaded implies view exists
    # unless the view was explicitly checked and found missing)
    if graph_loaded and view_exists is not False:
        return {"indicator": "green",
                "title": f"Digital Twin active — {count:,} triples",
                "count": count, "pending": False}

    # Red: never built and nothing exists
    if not project.last_build and not graph_loaded and not view_exists and not archive_exists:
        return {"indicator": "red", "title": "Digital Twin never built",
                "count": 0, "pending": False}

    # Orange: partial state
    parts = []
    if view_exists is False:
        parts.append("view missing")
    if not graph_loaded:
        parts.append("graph not loaded")
    title = "Digital Twin incomplete — " + ", ".join(parts) if parts else "Digital Twin partially available"
    return {"indicator": "orange", "title": title,
            "count": count, "pending": False}

def get_navbar_state(
    project,
    warehouse_id: str = "",
) -> Dict[str, Any]:
    """Build the complete navbar state in a single call.

    Combines project info, validation status, warehouse config, and
    cached digital-twin status so the client needs only one HTTP
    round-trip instead of four or five.
    """
    from app.html.project.service import get_project_info

    logger.debug("Building navbar state")
    project_data = get_project_info(project)

    validation = validate_status(project)

    dtwin = _compute_dtwin_indicator(project)

    return {
        "project": project_data,
        "validation": validation,
        "dtwin": dtwin,
        "warehouse": {"warehouse_id": warehouse_id},
    }
