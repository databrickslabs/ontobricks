@router.get("/load")
async def load_ontology(session_mgr: SessionManager = Depends(get_session_manager)):
    """Load ontology from session."""
    project = get_project(session_mgr)
    classes = project.get_classes()
    properties = project.get_properties()
    logger.debug("/ontology/load: ontology name=%s, classes=%s, properties=%s", project.ontology.get('name', ''), len(classes), len(properties))
    if classes:
        logger.debug("/ontology/load: first class=%s", classes[0].get('name', 'unknown'))

    # Normalize property domain/range to match actual class names (fix case mismatches)
    class_name_lookup = {c['name'].lower(): c['name'] for c in classes if c.get('name')}
    modified = False
    for prop in properties:
        for field in ('domain', 'range'):
            val = prop.get(field, '')
            if val and val not in class_name_lookup.values():
                canonical = class_name_lookup.get(val.lower())
                if canonical:
                    prop[field] = canonical
                    modified = True
    if modified:
        logger.info("/ontology/load: fixed domain/range case mismatches in properties")
        project.ontology['properties'] = properties
        project.save()

    ontology_name = project.ontology.get('name', '') or project.info.get('name', '')
    return {
        'success': True,
        'config': {
            'name': ontology_name,
            'base_uri': project.ontology.get('base_uri', ''),
            'description': project.ontology.get('description', ''),
            'classes': classes,
            'properties': properties
        }
    }

@router.post("/save")
async def save_ontology(request: Request, session_mgr: SessionManager = Depends(get_session_manager)):
    """Save ontology to session and clean up orphaned mappings."""
    data = await request.json()
    ontology_config = data.get('config', data)  # Support both wrapped and unwrapped
    
    # Ensure URIs are set
    ontology_config = ontology_service.ensure_uris(ontology_config)
    
    # Normalize property domain/range to match actual class names (fix case mismatches)
    class_name_lookup = {c['name'].lower(): c['name'] for c in ontology_config.get('classes', []) if c.get('name')}
    for prop in ontology_config.get('properties', []):
        for field in ('domain', 'range'):
            val = prop.get(field, '')
            if val and val not in class_name_lookup.values():
                canonical = class_name_lookup.get(val.lower())
                if canonical:
                    logger.debug("Normalizing property %s.%s: %r → %r", prop.get('name'), field, val, canonical)
                    prop[field] = canonical
    
    project = get_project(session_mgr)
    
    # Preserve existing constraints, swrl_rules, axioms if not provided in request
    existing_constraints = project.constraints
    existing_swrl_rules = project.swrl_rules
    existing_axioms = project.axioms
    
    # Get the new class and property URIs
    new_class_uris = {c.get('uri') for c in ontology_config.get('classes', []) if c.get('uri')}
    new_property_uris = {p.get('uri') for p in ontology_config.get('properties', []) if p.get('uri')}
    
    # Clean up orphaned entity mappings (mappings for classes that no longer exist)
    entity_mappings = project.get_entity_mappings()
    cleaned_entity_mappings = [m for m in entity_mappings if m.get('ontology_class') in new_class_uris]
    removed_entity_mappings = len(entity_mappings) - len(cleaned_entity_mappings)
    
    # Clean up orphaned relationship mappings (mappings for properties that no longer exist)
    rel_mappings = project.get_relationship_mappings()
    cleaned_rel_mappings = [m for m in rel_mappings if m.get('property') in new_property_uris]
    removed_rel_mappings = len(rel_mappings) - len(cleaned_rel_mappings)
    
    if removed_entity_mappings > 0 or removed_rel_mappings > 0:
        orphaned_uris = [m.get('ontology_class') for m in entity_mappings if m.get('ontology_class') not in new_class_uris]
        logger.warning(
            "Orphan cleanup: removing %d entity mappings (orphan URIs: %s) and %d rel mappings. "
            "New class URIs: %s",
            removed_entity_mappings, orphaned_uris,
            removed_rel_mappings, list(new_class_uris)[:10]
        )
    
    # Update mappings if any were removed
    if removed_entity_mappings > 0:
        project._data['assignment']['entities'] = cleaned_entity_mappings
    if removed_rel_mappings > 0:
        project._data['assignment']['relationships'] = cleaned_rel_mappings
    
    # Clear ALL cached content to ensure fresh regeneration
    project.clear_generated_content()
    
    project.ontology.update({
        'name': ontology_config.get('name', ''),
        'base_uri': ontology_config.get('base_uri', ''),
        'description': ontology_config.get('description', ''),
        'classes': ontology_config.get('classes', []),
        'properties': ontology_config.get('properties', []),
        # Preserve or update constraints/rules/axioms
        'constraints': ontology_config.get('constraints', existing_constraints),
        'swrl_rules': ontology_config.get('swrl_rules', existing_swrl_rules),
        'axioms': ontology_config.get('axioms', existing_axioms)
    })
    project.save()
    
    return {
        'success': True,
        'message': 'Ontology saved',
        'stats': ontology_service.get_ontology_stats(ontology_config),
        'mappings_cleaned': {
            'entity_mappings_removed': removed_entity_mappings,
            'relationship_mappings_removed': removed_rel_mappings
        }
    }

async def add_class(request: Request, session_mgr: SessionManager = Depends(get_session_manager)):
    """Add a class to the ontology."""
    data = await request.json()
    project = get_project(session_mgr)
    classes = project.get_classes()

    # Clear cached content
    project.clear_generated_content()
    project.save()
    return {'success': True, 'class': new_class}

    """Update a class in the ontology."""
    data = await request.json()
    project = get_project(session_mgr)
    classes = project.get_classes()
    class_uri = data.get('uri')

    return {'success': False, 'message': 'Class not found'}

@router.post("/class/delete")
async def delete_class(request: Request, session_mgr: SessionManager = Depends(get_session_manager)):
    """Delete a class from the ontology and its associated mappings."""
    data = await request.json()
    project = get_project(session_mgr)
    classes = project.get_classes()
    class_uri = data.get('uri')
    
    original_len = len(classes)
    classes = [c for c in classes if c.get('uri') != class_uri]
    
    if len(classes) < original_len:
        project.ontology['classes'] = classes
        
        # Cascade delete: Remove associated entity mapping
        entity_mappings = project.get_entity_mappings()
        original_mapping_len = len(entity_mappings)
        entity_mappings = [m for m in entity_mappings if m.get('ontology_class') != class_uri]
        if len(entity_mappings) < original_mapping_len:
            project._data['assignment']['entities'] = entity_mappings
        
        # Clear cached content
        project.clear_generated_content()
        
        project.save()
        return {'success': True, 'mapping_removed': len(entity_mappings) < original_mapping_len}
    return {'success': False, 'message': 'Class not found'}

async def generate_owl_endpoint(request: Request, session_mgr: SessionManager = Depends(get_session_manager)):
    """Generate OWL from ontology configuration."""
    data = await request.json()
    if not data:
        return {'success': False, 'message': 'No data provided'}
    
    try:
        project = get_project(session_mgr)
        constraints = data.get('constraints') or project.constraints
        swrl_rules = data.get('swrl_rules') or project.swrl_rules

        owl_content = ontology_service.generate_owl(data, constraints, swrl_rules, axioms)
        project.generated['owl'] = owl_content
        project.save()
        
        return {'success': True, 'owl': owl_content, 'format': 'turtle'}
    except Exception as e:
        logger.exception("OWL generation failed: %s", e)
        return {'success': False, 'message': str(e)}

@router.post("/import-owl")
async def import_owl(request: Request, session_mgr: SessionManager = Depends(get_session_manager)):
    """Import ontology from OWL/TTL content."""
    data = await request.json()
    owl_content = data.get('content', '')
    
    if not owl_content:
        return {'success': False, 'message': 'No OWL content provided'}
    
    try:
        result = ontology_service.parse_owl(owl_content, extract_advanced=True)

        project = get_project(session_mgr)
        ontology_service.apply_parsed_owl_to_project(
            project,
            ontology_info,
            classes,
            properties,
            constraints,
            swrl_rules,
            axioms,
            name_fallback_to_project=True,
        )
        return ontology_service.build_import_owl_success_payload(
            project, classes, properties, constraints
        )
    except Exception as e:
        logger.exception("OWL import failed: %s", e)
        return {'success': False, 'message': str(e)}

@router.post("/dataquality/save")
async def save_shape(request: Request, session_mgr: SessionManager = Depends(get_session_manager)):
    """Add or update a SHACL data-quality shape."""
    try:
        data = await request.json()
        shape_data = data.get("shape", {})
        shape_id = shape_data.get("id", "")

        error = ontology_service.validate_shape(shape_data)
        if error:
            return {"success": False, "message": error}

        project = get_project(session_mgr)
        shapes = project.ensure_shacl_shapes()

        if shape_id and any(s["id"] == shape_id for s in shapes):
            shapes = SHACLService.update_shape(shapes, shape_id, shape_data)
        else:
            new_shape = SHACLService.create_shape(
                category=shape_data.get("category", "conformance"),
                target_class=shape_data.get("target_class", ""),
                target_class_uri=shape_data.get("target_class_uri", ""),
                property_path=shape_data.get("property_path", ""),
                property_uri=shape_data.get("property_uri", ""),
                shacl_type=shape_data.get("shacl_type", "sh:minCount"),
                parameters=shape_data.get("parameters", {}),
                severity=shape_data.get("severity", "sh:Violation"),
                message=shape_data.get("message", ""),
                label=shape_data.get("label", ""),
                enabled=shape_data.get("enabled", True),
            )
            shapes.append(new_shape)

        project.shacl_shapes = shapes
        project.save()
        return {"success": True, "message": "Shape saved", "shapes": shapes}
    except Exception as e:
        logger.exception("Save shape failed: %s", e)
        return {"success": False, "message": str(e)}

@router.post("/load-owl-file")
async def load_owl_file(
    request: Request,
    session_mgr: SessionManager = Depends(get_session_manager),
    settings: Settings = Depends(get_settings)
):
    """Load and parse an OWL file from Unity Catalog."""
    data = await request.json()
    catalog, schema, volume, filename = data.get('catalog'), data.get('schema'), data.get('volume'), data.get('filename')
    
    if not all([catalog, schema, volume, filename]):
        return {'success': False, 'message': 'Missing required fields'}
    
    try:
        project = get_project(session_mgr)
        host, token = get_databricks_host_and_token(project, settings)
        uc_service = VolumeFileService(host=host, token=token)
        file_path = f"/Volumes/{catalog}/{schema}/{volume}/{filename}"
        
        success, owl_content, message = uc_service.read_file(file_path)
        if not success:
            return {'success': False, 'message': message}
        
        result = ontology_service.parse_owl(owl_content, extract_advanced=True)
        ontology_info, classes, properties, constraints, swrl_rules, axioms = result

        ontology_service.apply_parsed_owl_to_project(
            project,
            ontology_info,
            classes,
            properties,
            constraints,
            swrl_rules,
            axioms,
            name_fallback_to_project=False,
        )
        return ontology_service.build_load_owl_file_success_payload(
            project, classes, properties, constraints, swrl_rules, axioms
        )
    except Exception as e:
        logger.exception("Load OWL file failed: %s", e)
        return {'success': False, 'message': str(e)}

@router.post("/update-relationship-references")
async def update_relationship_references(request: Request, session_mgr: SessionManager = Depends(get_session_manager)):
    """Update references when a relationship is renamed."""
    data = await request.json()
    old_name, new_name = data.get('old_name'), data.get('new_name')
    
    if not old_name or not new_name:
        return {'success': False, 'message': 'Both old_name and new_name are required'}
    
    project = get_project(session_mgr)
    updates = {'mappings_updated': 0, 'constraints_updated': 0, 'axioms_updated': 0}
    
    # Update mapping references
    for rel_mapping in project.get_relationship_mappings():
        if rel_mapping.get('property_label') == old_name:
            rel_mapping['property_label'] = new_name
            updates['mappings_updated'] += 1
    
    # Update constraints
    for constraint in project.constraints:
        if constraint.get('property') == old_name:
            constraint['property'] = new_name
            updates['constraints_updated'] += 1
    
    # Update axioms
    for axiom in project.axioms:
        if axiom.get('property') == old_name:
            axiom['property'] = new_name
            updates['axioms_updated'] += 1
    
    project.save()
    
    total = sum(updates.values())
    return {'success': True, 'message': f'Updated {total} references', 'updates': updates}

@router.post("/wizard/generate-async")
async def generate_ontology_async(
    request: Request,
    session_mgr: SessionManager = Depends(get_session_manager),
    settings: Settings = Depends(get_settings)
):
    """Start async ontology generation via the agent loop and return task ID."""
    import threading
    from agents.agent_owl_generator import run_agent

    data = await request.json()
    metadata = data.get('metadata', {})
    guidelines = data.get('guidelines', '')
    options = data.get('options', {})
    documents = data.get('documents', [])

    tables_count = len(metadata.get('tables', []))

    project = get_project(session_mgr)
    host, token = get_databricks_host_and_token(project, settings)

    if not host or not token:
        return {'success': False, 'message': 'Databricks credentials not configured'}

    llm_endpoint = project.info.get('llm_endpoint', '')
    if not llm_endpoint:
        return {'success': False, 'message': 'No LLM serving endpoint configured. Please set it in Project Settings.'}

    base_uri = project.ontology.get('base_uri') or project.ontology.get('info', {}).get('base_uri') or DEFAULT_BASE_URI

    # Selected table names (from the metadata the client already filtered)
    selected_tables = [
        t.get('full_name') or t.get('name') for t in metadata.get('tables', [])
    ]

    tm = get_task_manager()
    task = tm.create_task(
        name=f"Generate Ontology ({tables_count} tables)" if tables_count else "Generate Ontology (guidelines only)",
        task_type="ontology_generation",
        steps=[
            {'name': 'init', 'description': 'Initializing agent'},
            {'name': 'gather', 'description': 'Gathering context (metadata & documents)'},
            {'name': 'generate', 'description': 'Generating ontology with AI'},
            {'name': 'process', 'description': 'Processing results'},
            {'name': 'finalize', 'description': 'Finalizing'},
        ]
    )

    def run_generation():
        try:
            tm.start_task(task.id, "Starting agent…")

            def on_step(msg: str):
                tm.update_progress(task.id, task.progress, msg)

            agent_result = run_agent(
                host=host,
                token=token,
                endpoint_name=llm_endpoint,
                registry=dict(project.registry),
                metadata=metadata,
                guidelines=guidelines,
                options=options,
                base_uri=base_uri,
                project_name=project.info.get('name', ''),
                project_folder=project.project_folder,
                selected_tables=selected_tables,
                selected_docs=documents,
                on_step=on_step,
            )

            if not agent_result.success:
                tm.fail_task(task.id, agent_result.error or "Agent did not produce output")
                return

            tm.advance_step(task.id, "Processing results…")
            owl_content = ontology_service.clean_owl_output(agent_result.owl_content)
            stats = ontology_service.calculate_owl_stats(owl_content)

            tm.advance_step(task.id, "Finalizing…")
            tm.complete_task(
                task.id,
                result={
                    'owl_content': owl_content,
                    'stats': stats,
                    'agent_steps': [
                        {'type': s.step_type, 'tool': s.tool_name,
                         'content': s.content, 'ms': s.duration_ms}
                        for s in agent_result.steps
                    ],
                    'agent_iterations': agent_result.iterations,
                    'agent_usage': agent_result.usage,
                },
                message=(
                    f"Generated {stats.get('classes', 0)} classes, "
                    f"{stats.get('properties', 0)} properties "
                    f"({agent_result.iterations} agent iterations)"
                ),
            )

        except Exception as e:
            logger.exception("Wizard async: Ontology generation failed: %s", e)
            tm.fail_task(task.id, str(e))

    thread = threading.Thread(target=run_generation, daemon=True)
    thread.start()

    return {
        'success': True,
        'task_id': task.id,
        'message': 'Agent task started'
    }

@router.post("/auto-assign-icons")
async def auto_assign_icons(
    request: Request,
    session_mgr: SessionManager = Depends(get_session_manager),
    settings: Settings = Depends(get_settings)
):
    """Use the auto-icon-assign agent to suggest emoji icons for entity names."""
    from agents.agent_auto_icon_assign import run_agent as run_icon_agent

    try:
        data = await request.json()
        entity_names = data.get('entity_names', [])

        if not entity_names:
            return {'success': False, 'message': 'No entity names provided'}

        project = get_project(session_mgr)
        host, token = get_databricks_host_and_token(project, settings)

        if not host or not token:
            return {'success': False, 'message': 'Databricks credentials not configured'}

        llm_endpoint = project.info.get('llm_endpoint', '')
        if not llm_endpoint:
            return {'success': False, 'message': 'No LLM serving endpoint configured. Please set it in Project Settings.'}

        # Build ontology context for the agent
        classes = project.get_classes()
        properties = project.get_properties()
        ontology_ctx = {
            "entities": [
                {
                    "name": c.get("name", ""),
                    "uri": c.get("uri", ""),
                    "attributes": [
                        dp.get("name", "") for dp in c.get("dataProperties", [])
                    ],
                }
                for c in classes
            ],
            "relationships": [
                {
                    "name": p.get("name", ""),
                    "domain": p.get("domain", ""),
                    "range": p.get("range", ""),
                }
                for p in properties
                if p.get("type") in ("ObjectProperty", "owl:ObjectProperty", None)
            ],
        }

        metadata = project._data.get('metadata', {})

        logger.info("AutoIcons: launching agent for %d entities", len(entity_names))

        agent_result = run_icon_agent(
            host=host,
            token=token,
            endpoint_name=llm_endpoint,
            entity_names=entity_names,
            metadata=metadata,
            ontology=ontology_ctx,
        )

        if not agent_result.success:
            logger.warning("AutoIcons: agent failed — %s", agent_result.error)
            return {'success': False, 'message': agent_result.error or 'Agent failed to assign icons'}

        # Build a case-insensitive lookup so small LLM variations don't cause misses
        normalized_map = {}
        for key, emoji in agent_result.icons.items():
            normalized_map[key] = emoji
            normalized_map[key.lower()] = emoji

        final_map = {}
        for name in entity_names:
            emoji = normalized_map.get(name) or normalized_map.get(name.lower())
            if emoji:
                final_map[name] = emoji

        missing = [n for n in entity_names if n not in final_map]
        if missing:
            logger.warning("AutoIcons: No icon for: %s", missing)

        logger.info(
            "AutoIcons: agent completed — %d/%d icons assigned in %d iterations",
            len(final_map), len(entity_names), agent_result.iterations,
        )

        return {
            'success': True,
            'icons': final_map,
            'agent_iterations': agent_result.iterations,
        }

    except Exception as e:
        logger.exception("AutoIcons: Failed: %s", e)
        return {'success': False, 'message': str(e)}

@router.post("/assistant/chat")
async def ontology_assistant_chat(
    request: Request,
    session_mgr: SessionManager = Depends(get_session_manager),
    settings: Settings = Depends(get_settings)
):
    """Process a single chat turn with the ontology assistant agent.

    Expects JSON body:
        {
            "message": "Remove the entity Customer",
            "history": [...]   // optional prior conversation messages
        }

    Returns:
        {
            "success": true/false,
            "reply": "...",
            "ontology_changed": true/false,
            "config": { classes, properties, ... }   // returned when ontology was modified
        }
    """
    from agents.agent_ontology_assistant import run_agent as run_assistant

    data = await request.json()
    user_message = data.get("message", "").strip()
    history = data.get("history", [])

    if not user_message:
        return {"success": False, "message": "No message provided"}

    project = get_project(session_mgr)
    host, token = get_databricks_host_and_token(project, settings)

    if not host or not token:
        return {"success": False, "message": "Databricks credentials not configured"}

    llm_endpoint = project.info.get("llm_endpoint", "")
    if not llm_endpoint:
        return {"success": False, "message": "No LLM serving endpoint configured. Please set it in Project Settings."}

    classes = list(project.get_classes())
    properties = list(project.get_properties())
    base_uri = project.ontology.get("base_uri") or DEFAULT_BASE_URI

    logger.info("OntologyAssistant: user_message=%s, classes=%d, properties=%d",
                user_message[:80], len(classes), len(properties))

    try:
        agent_result = run_assistant(
            host=host,
            token=token,
            endpoint_name=llm_endpoint,
            classes=classes,
            properties=properties,
            base_uri=base_uri,
            user_message=user_message,
            conversation_history=history,
        )
    except Exception as exc:
        logger.exception("OntologyAssistant: agent failed: %s", exc)
        return {"success": False, "message": f"Assistant error: {exc}"}

    if not agent_result.success:
        return {"success": False, "message": agent_result.error or "Assistant failed"}

    response = {
        "success": True,
        "reply": agent_result.reply,
        "ontology_changed": agent_result.ontology_changed,
    }

    if agent_result.ontology_changed:
        ontology_config = {
            "name": project.ontology.get("name", ""),
            "base_uri": base_uri,
            "description": project.ontology.get("description", ""),
            "classes": agent_result.classes,
            "properties": agent_result.properties,
        }
        ontology_config = ontology_service.ensure_uris(ontology_config)

        new_class_uris = {c.get("uri") for c in ontology_config["classes"] if c.get("uri")}
        new_property_uris = {p.get("uri") for p in ontology_config["properties"] if p.get("uri")}
        entity_mappings = project.get_entity_mappings()
        cleaned_entity_mappings = [m for m in entity_mappings if m.get("ontology_class") in new_class_uris]
        rel_mappings = project.get_relationship_mappings()
        cleaned_rel_mappings = [m for m in rel_mappings if m.get("property") in new_property_uris]

        if len(cleaned_entity_mappings) < len(entity_mappings):
            project._data["assignment"]["entities"] = cleaned_entity_mappings
        if len(cleaned_rel_mappings) < len(rel_mappings):
            project._data["assignment"]["relationships"] = cleaned_rel_mappings

        project.clear_generated_content()
        project.ontology.update({
            "classes": ontology_config["classes"],
            "properties": ontology_config["properties"],
        })
        project.save()

        response["config"] = {
            "name": project.ontology.get("name", ""),
            "base_uri": base_uri,
            "description": project.ontology.get("description", ""),
            "classes": ontology_config["classes"],
            "properties": ontology_config["properties"],
        }
        logger.info("OntologyAssistant: ontology saved — classes=%d, properties=%d",
                     len(ontology_config["classes"]), len(ontology_config["properties"]))

    return response

@router.post("/assistant/invoke")
async def ontology_assistant_invoke(
    request: Request,
    session_mgr: SessionManager = Depends(get_session_manager),
    settings: Settings = Depends(get_settings),
):
    """Invoke the Ontology Assistant via the MLflow ResponsesAgent interface.

    Accepts the OpenAI Responses-compatible schema used by the Databricks
    Agent Framework.  The caller can either supply Databricks credentials in
    ``custom_inputs`` or let the route fill them from the active session.

    Expects JSON body (ResponsesAgentRequest)::

        {
            "input": [
                {"role": "user", "content": "Add an entity called Vehicle"}
            ],
            "custom_inputs": {          // optional overrides
                "host": "...",
                "token": "...",
                "endpoint_name": "..."
            }
        }

    Returns a ``ResponsesAgentResponse`` with ``custom_outputs`` containing
    the mutated ontology when changes were made.
    """
    from agents.agent_ontology_assistant import OntologyAssistantResponsesAgent
    from mlflow.types.responses import ResponsesAgentRequest as RAReq

    data = await request.json()

    project = get_project(session_mgr)
    host, token = get_databricks_host_and_token(project, settings)
    llm_endpoint = project.info.get("llm_endpoint", "")
    base_uri = project.ontology.get("base_uri") or DEFAULT_BASE_URI

    custom_inputs = data.get("custom_inputs", {})
    custom_inputs.setdefault("host", host)
    custom_inputs.setdefault("token", token)
    custom_inputs.setdefault("endpoint_name", llm_endpoint)
    custom_inputs.setdefault("base_uri", base_uri)
    custom_inputs.setdefault("classes", list(project.get_classes()))
    custom_inputs.setdefault("properties", list(project.get_properties()))
    data["custom_inputs"] = custom_inputs

    if not custom_inputs.get("host") or not custom_inputs.get("token"):
        return {"error": "Databricks credentials not configured"}
    if not custom_inputs.get("endpoint_name"):
        return {"error": "No LLM serving endpoint configured."}

    try:
        agent = OntologyAssistantResponsesAgent()
        ra_request = RAReq(**data)
        response = agent.predict(ra_request)

        co = response.custom_outputs or {}
        if co.get("ontology_changed"):
            ontology_config = {
                "name": project.ontology.get("name", ""),
                "base_uri": base_uri,
                "description": project.ontology.get("description", ""),
                "classes": co.get("classes", []),
                "properties": co.get("properties", []),
            }
            ontology_config = ontology_service.ensure_uris(ontology_config)
            project.clear_generated_content()
            project.ontology.update({
                "classes": ontology_config["classes"],
                "properties": ontology_config["properties"],
            })
            project.save()

        return response.model_dump()
    except Exception as exc:
        logger.exception("ontology_assistant_invoke failed: %s", exc)
        return {"error": str(exc)}
