Use :class:`Ontology` with a :class:`~app.objects.session.ProjectSession` for
operations that persist to the session; use static methods for pure transforms.

    fetch_and_parse_fibo,
    fetch_and_parse_iof,

    from app.objects.session.project_session import ProjectSession

_INDUSTRY_FETCH = {
    "fibo": fetch_and_parse_fibo,
    "cdisc": fetch_and_parse_cdisc,

    @staticmethod
    def ensure_uris(config: Dict[str, Any]) -> Dict[str, Any]:
        """Ensure all classes and properties have URIs.
        
        Args:
            config: Ontology configuration dict
            
        Returns:
            dict: Configuration with URIs ensured
        """
        base_uri = config.get('base_uri', 'http://example.org/')
        if not base_uri.endswith('#') and not base_uri.endswith('/'):
            base_uri = base_uri + '#'
        
        for cls in config.get('classes', []):
            if not cls.get('uri') and cls.get('name'):
                cls['uri'] = base_uri + cls['name']
            if not cls.get('localName') and cls.get('name'):
                cls['localName'] = cls['name']
        
        for prop in config.get('properties', []):
            if not prop.get('uri') and prop.get('name'):
                prop['uri'] = base_uri + prop['name']
            if not prop.get('localName') and prop.get('name'):
                prop['localName'] = prop['name']
        
        return config

    @staticmethod
    def validate_constraint(constraint: Dict[str, Any]) -> Optional[str]:
        """Validate constraint and return error message if invalid.
        
        Args:
            constraint: Constraint data
            
        Returns:
            str: Error message if invalid, None if valid
        """
        constraint_type = constraint.get('type')
        if not constraint_type:
            return 'Constraint type is required'
        
        property_characteristics = ['functional', 'inverseFunctional', 'transitive', 
                                    'symmetric', 'asymmetric', 'reflexive', 'irreflexive']
        cardinality_constraints = ['minCardinality', 'maxCardinality', 'exactCardinality']
        value_constraints = ['valueCheck', 'entityValueCheck', 'entityLabelCheck', 
                            'attributeConstraint', 'globalRule']
        
        if constraint_type in cardinality_constraints:
            if not constraint.get('property'):
                return 'Relationship (property) is required for cardinality constraints'
            if constraint.get('cardinalityValue') is None:
                return 'Cardinality value is required'
        elif constraint_type in value_constraints:
            if constraint_type != 'globalRule' and not constraint.get('className'):
                return 'Entity (className) is required for value constraints'
        elif constraint_type in property_characteristics:
            if not constraint.get('property'):
                return 'Property is required for property characteristics'
        
        return None

    @staticmethod
    def validate_shape(shape: Dict[str, Any]) -> Optional[str]:
        """Validate a SHACL shape dict, return error message or None."""
        category = shape.get("category", "")
        if category not in QUALITY_CATEGORIES:
            return f"Invalid category '{category}'. Must be one of: {', '.join(QUALITY_CATEGORIES)}"
    
        shacl_type = shape.get("shacl_type", "")
        if not shacl_type:
            return "shacl_type is required"
    
        if shacl_type not in ("sh:sparql", "sh:closed"):
            if not shape.get("target_class") and not shape.get("target_class_uri"):
                pass
            if not shape.get("property_path") and not shape.get("property_uri"):
                return "A property path or URI is required for this constraint type"
    
        params = shape.get("parameters", {})
        if shacl_type in ("sh:minCount", "sh:maxCount"):
            for key in ("sh:minCount", "sh:maxCount"):
                if key in params:
                    try:
                        int(params[key])
                    except (ValueError, TypeError):
                        return f"{key} must be an integer"
    
        return None

    @staticmethod
    def parse_owl(content, extract_advanced=True):
        """Parse OWL content and return structured data.
        
        Args:
            content: OWL/Turtle content
            extract_advanced: If True, also extract constraints, SWRL rules, and axioms
            
        Returns:
            tuple: (ontology_info, classes, properties) or 
                   (ontology_info, classes, properties, constraints, swrl_rules, axioms) if extract_advanced=True
        """
        parser = OntologyParser(content)
        ontology_info = parser.get_ontology_info()
        classes = parser.get_classes()
        properties = parser.get_properties()
        
        if extract_advanced:
            constraints = parser.get_constraints()
            swrl_rules = parser.get_swrl_rules()
            axioms = parser.get_axioms()
            return ontology_info, classes, properties, constraints, swrl_rules, axioms
        
        return ontology_info, classes, properties

    def import_industry_ontology(self, kind: IndustryKind, domain_keys: List[str]) -> Dict[str, Any]:
        """Fetch industry modules, merge, parse, persist to project session.
    
        Returns the same dict shape as the former /import-fibo|cdisc|iof handlers.
        """
        if not domain_keys:
            return {"success": False, "message": _INDUSTRY_EMPTY_MESSAGE[kind]}
    
        try:
            result = _INDUSTRY_FETCH[kind](domain_keys)
    
            if not result["success"]:
                return {
                    "success": False,
                    "message": result["message"],
                    "failed": result.get("failed", []),
                }
    
            info = result["ontology_info"]
            if kind == "cdisc":
                ont_name = info.get("label", "CDISC")
                base_uri = info.get("uri", "http://rdf.cdisc.org/")
                desc_prefix = "CDISC Foundational Standards in RDF – "
            elif kind == "fibo":
                ont_name = info.get("name", "FIBO")
                base_uri = info.get("uri", "https://spec.edmcouncil.org/fibo/ontology/")
                desc_prefix = "Financial Industry Business Ontology (FIBO) – "
            else:
                ont_name = info.get("name", "IOF")
                base_uri = info.get("uri", "https://spec.industrialontologies.org/ontology/")
                desc_prefix = "Industrial Ontologies Foundry (IOF) – "
    
            self._s.ontology.update({
                "name": ont_name,
                "base_uri": base_uri,
                "description": desc_prefix + ", ".join(domain_keys),
                "classes": result["classes"],
                "properties": result["properties"],
                "constraints": result["constraints"],
                "swrl_rules": result["swrl_rules"],
                "axioms": result["axioms"],
            })
            self._s.save()
    
            return {
                "success": True,
                "message": result["message"],
                "stats": result["stats"],
                "failed": result["failed"],
            }
        except Exception as exc:
            logger.exception("%s import failed: %s", _INDUSTRY_LOG_LABEL[kind], exc)
            return {"success": False, "message": str(exc)}

    @staticmethod
    def generate_ontology_with_llm(
        host: str,
        token: str,
        endpoint_name: str,
        metadata: Dict,
        guidelines: str,
        options: Dict,
        base_uri: str,
        documents: Optional[list] = None
    ) -> tuple:
        """Generate an ontology from metadata using LLM.
        
        Args:
            host: Databricks workspace host
            token: Databricks access token
            endpoint_name: Model serving endpoint name
            metadata: Project metadata with tables and columns
            guidelines: User-provided guidelines for ontology generation
            options: Generation options (includeDataProperties, etc.)
            base_uri: Base URI for the ontology
            documents: Optional list of document filenames selected as context
            
        Returns:
            Tuple of (owl_content, stats)
        """
        import requests
        
        # Build the schema description from metadata
        schema_desc = Ontology._build_schema_description(metadata)
        
        # Build the system prompt - OPTIMIZED: concise to reduce token count
        system_prompt = """You are an expert ontology engineer creating OWL ontologies in Turtle format.
    
    NAMING RULES (CRITICAL - NO EXCEPTIONS):
    - Classes: PascalCase (Customer, SalesOrder)
    - Properties: lowerCamelCase (hasName, firstName)
    - NO spaces/underscores/hyphens/accents in local names
    - Convert: "street_address" → streetAddress, "Sales Order" → SalesOrder
    
    PROPERTY TYPES:
    - owl:DatatypeProperty + xsd:type for attributes (string/integer/date values)
    - owl:ObjectProperty for relationships between classes only
    
    RELATIONSHIP RULES (CRITICAL):
    - NEVER create bidirectional relationships: between any two classes A and B, create at most ONE ObjectProperty (either A→B or B→A, not both).
    - For example, if you create :hasCustomer (Order → Customer), do NOT also create :hasOrder (Customer → Order).
    - Choose the most natural direction that best represents the domain semantics.
    - Each pair of classes should be connected by at most one ObjectProperty.
    
    Output ONLY valid Turtle. No markdown. Start with @prefix declarations."""
    
        # Build the user prompt - OPTIMIZED: concise
        include_attrs = options.get('includeDataProperties', True)
        include_rels = options.get('includeRelationships', True)
        
        docs_section = ""
        if documents:
            docs_section = "\nReference documents (use as additional domain context): " + ", ".join(documents)
    
        user_prompt = f"""Schema:
    {schema_desc}
    Base URI: {base_uri}
    Guidelines: {guidelines if guidelines else "Create domain ontology from schema."}{docs_section}
    
    Generate OWL with:{' DatatypeProperties for attributes,' if include_attrs else ''}{' ObjectProperties for relationships,' if include_rels else ''} rdfs:labels.
    Remember: NO spaces in local names (streetAddress not "street address").
    IMPORTANT: For ObjectProperties, create only ONE relationship per pair of classes (no bidirectional or duplicate relationships). Pick the most natural direction."""
    
        # Call the LLM endpoint
        url = f"{host.rstrip('/')}/serving-endpoints/{endpoint_name}/invocations"
        headers = {
            "Authorization": f"Bearer {token}",
            "Content-Type": "application/json"
        }
        
        # Calculate appropriate max_tokens - optimized for speed
        tables_count = len(metadata.get('tables', []))
        total_columns = sum(len(t.get('columns', [])) for t in metadata.get('tables', []))
        # More conservative allocation: ~80 tokens per class, ~25 per property
        max_tokens = min(3000, 500 + (tables_count * 100) + (total_columns * 20))
        
        payload = {
            "messages": [
                {"role": "system", "content": system_prompt},
                {"role": "user", "content": user_prompt}
            ],
            "max_tokens": max_tokens,
            "temperature": 0.1  # Very low for consistent, fast output
        }
        
        # Minimal logging for performance
        import time
        start_time = time.time()
        
        logger.info(
            "LLM call: endpoint=%s, tables=%s, cols=%s, max_tokens=%s",
            endpoint_name, tables_count, total_columns, max_tokens
        )
        
        # Use session for connection reuse with retry
        session = requests.Session()
        max_retries = 2
        last_error = None
        response = None
        
        for attempt in range(1, max_retries + 1):
            try:
                response = session.post(url, headers=headers, json=payload, timeout=180)
                break  # Success
            except requests.exceptions.ReadTimeout as e:
                last_error = e
                elapsed = time.time() - start_time
                logger.warning(
                    "Timeout on attempt %s/%s after %.1fs",
                    attempt, max_retries, elapsed
                )
                if attempt < max_retries:
                    logger.info("Retrying LLM call")
                else:
                    raise
        
        elapsed = time.time() - start_time
        logger.info("LLM response: status=%s in %.1fs", response.status_code, elapsed)
        response.raise_for_status()
        
        result = response.json()
        
        # Extract the generated content
        if 'choices' in result:
            owl_content = result['choices'][0].get('message', {}).get('content', '')
            usage = result.get('usage', {})
            if usage:
                logger.info("Tokens: prompt=%s, completion=%s", usage.get('prompt_tokens', '?'), usage.get('completion_tokens', '?'))
        elif 'predictions' in result:
            owl_content = result['predictions'][0] if result['predictions'] else ''
        else:
            owl_content = str(result)
        
        logger.info("Generated OWL: %s chars", len(owl_content))
        
        # Clean up the output (remove markdown fences if present)
        owl_content = Ontology.clean_owl_output(owl_content)
        
        # Calculate stats by parsing the generated OWL
        stats = Ontology.calculate_owl_stats(owl_content)
        logger.info("Wizard stats: %s", stats)
        
        return owl_content, stats



    @staticmethod
    def _build_schema_description(metadata: Dict) -> str:
        """Build a human-readable schema description from metadata."""
        lines = []
        tables = metadata.get('tables', [])
        
        for table in tables:
            table_name = table.get('full_name') or table.get('name', 'unknown')
            table_desc = table.get('comment') or table.get('description', '')
            
            lines.append(f"Table: {table_name}")
            if table_desc:
                lines.append(f"  Description: {table_desc}")
            
            columns = table.get('columns', [])
            if columns:
                lines.append("  Columns:")
                for col in columns:
                    col_name = col.get('name', '')
                    col_type = col.get('type', col.get('data_type', ''))
                    col_comment = col.get('comment', '')
                    
                    col_line = f"    - {col_name} ({col_type})"
                    if col_comment:
                        col_line += f": {col_comment}"
                    lines.append(col_line)
            
            lines.append("")
        
        return "\n".join(lines)

    @staticmethod
    def _fix_local_names(content: str) -> str:
        """Fix local names with spaces, underscores, or hyphens in Turtle content.
        
        Converts patterns like:
        - :street address → :streetAddress
        - :Street Address → :StreetAddress  
        - :first_name → :firstName
        - :customer-id → :customerId
        """
        import re
        
        # Turtle/RDF keywords that should never be joined with previous word
        TURTLE_KEYWORDS = {'a', 'rdf', 'rdfs', 'owl', 'xsd', 'xml', 'true', 'false'}
        
        def to_camel_case(words, is_pascal):
            """Convert list of words to camelCase or PascalCase."""
            if not words:
                return ''
            if is_pascal:
                return ''.join(word.capitalize() for word in words if word)
            else:
                result = words[0].lower()
                for word in words[1:]:
                    if word:
                        result += word.capitalize()
                return result
        
        def fix_underscore_name(match):
            """Convert snake_case or kebab-case to camelCase/PascalCase."""
            prefix = match.group(1)
            name = match.group(2)
            
            words = re.split(r'[_-]+', name)
            if len(words) <= 1:
                return match.group(0)
            
            is_pascal = words[0] and words[0][0].isupper()
            return prefix + to_camel_case(words, is_pascal)
        
        # Step 1: Fix snake_case and kebab-case (safe - these are always wrong in Turtle)
        underscore_pattern = r'(?<![a-zA-Z])(:)([a-zA-Z][a-zA-Z0-9]*(?:[_-][a-zA-Z][a-zA-Z0-9]*)+)'
        content = re.sub(underscore_pattern, fix_underscore_name, content)
        
        # Step 2: Fix spaces in local names using a token-based approach
        # Process the content by finding all :LocalName patterns and checking for spaces
        
        def fix_line(line):
            """Fix a single line, handling multiple :LocalName patterns."""
            # Skip comment lines and prefix declarations
            stripped = line.strip()
            if stripped.startswith('#') or stripped.startswith('@prefix') or stripped.startswith('@base'):
                return line
            
            result = []
            i = 0
            
            while i < len(line):
                # Look for : that starts a local name (not part of a prefix like rdf:)
                if line[i] == ':':
                    # Check if this is a prefix usage (word before :) or a local name (start or after whitespace)
                    is_local_name = (i == 0 or line[i-1] in ' \t;.,()[]')
                    
                    if is_local_name:
                        # Extract the local name which might have spaces
                        j = i + 1
                        
                        # Skip if nothing follows
                        if j >= len(line):
                            result.append(line[i])
                            i += 1
                            continue
                        
                        # Collect words that should be part of the local name
                        words = []
                        current_word = ''
                        
                        while j < len(line):
                            ch = line[j]
                            
                            if ch.isalnum():
                                current_word += ch
                                j += 1
                            elif ch == ' ' and current_word:
                                # Space found - check if next word should be joined
                                # Look ahead to see what comes next
                                k = j + 1
                                while k < len(line) and line[k] == ' ':
                                    k += 1
                                
                                if k < len(line):
                                    # Check what's next
                                    next_char = line[k]
                                    
                                    # If next is a letter, check if it's a keyword or prefix
                                    if next_char.isalpha():
                                        # Extract the next word
                                        next_word_end = k
                                        while next_word_end < len(line) and line[next_word_end].isalnum():
                                            next_word_end += 1
                                        next_word = line[k:next_word_end]
                                        
                                        # Check if followed by : (making it a prefix)
                                        after_next = next_word_end
                                        while after_next < len(line) and line[after_next] == ' ':
                                            after_next += 1
                                        
                                        is_prefix = after_next < len(line) and line[after_next] == ':'
                                        is_keyword = next_word.lower() in TURTLE_KEYWORDS
                                        
                                        # Don't join if it's a prefix usage or keyword
                                        if is_prefix or is_keyword or len(next_word) == 1:
                                            # End of local name
                                            words.append(current_word)
                                            break
                                        else:
                                            # This word should be joined
                                            words.append(current_word)
                                            current_word = ''
                                            j = k  # Move to start of next word
                                    else:
                                        # Next is not a letter - end of local name
                                        words.append(current_word)
                                        break
                                else:
                                    # End of line
                                    words.append(current_word)
                                    break
                            else:
                                # Not alphanumeric or space - end of local name
                                if current_word:
                                    words.append(current_word)
                                break
                        
                        # If we collected words, convert to camelCase
                        if words:
                            is_pascal = words[0] and words[0][0].isupper()
                            fixed_name = to_camel_case(words, is_pascal)
                            result.append(':')
                            result.append(fixed_name)
                            i = j
                        else:
                            result.append(line[i])
                            i += 1
                    else:
                        result.append(line[i])
                        i += 1
                else:
                    result.append(line[i])
                    i += 1
            
            return ''.join(result)
        
        lines = content.split('\n')
        fixed_lines = [fix_line(line) for line in lines]
        
        return '\n'.join(fixed_lines)

    @staticmethod
    def clean_owl_output(content: str) -> str:
        """Clean up LLM output to extract valid Turtle content."""
        content = content.strip()
    
        # Remove markdown code fences
        if '```' in content:
            import re
            m = re.search(r'```(?:turtle|ttl|sparql|rdf)?\s*\n(.*?)```', content, re.DOTALL)
            if m:
                content = m.group(1).strip()
            elif content.startswith('```'):
                lines = content.split('\n')
                lines = lines[1:]
                if lines and lines[-1].strip() == '```':
                    lines = lines[:-1]
                content = '\n'.join(lines)
    
        content = content.strip()
    
        # Strip any natural-language preamble before the first @prefix or @base
        prefix_idx = content.find('@prefix')
        base_idx = content.find('@base')
        candidates = [i for i in (prefix_idx, base_idx) if i > 0]
        if candidates:
            content = content[min(candidates):]
    
        content = content.strip()
    
        # Fix any local names with spaces, underscores, or hyphens
        content = Ontology._fix_local_names(content)

        return content

    @staticmethod
    def calculate_owl_stats(owl_content: str) -> Dict:
        """Calculate statistics from OWL content."""
        stats = {
            'classes': 0,
            'properties': 0,
            'dataProperties': 0
        }
        
        try:
            # Count owl:Class declarations
            stats['classes'] = owl_content.count('a owl:Class') + owl_content.count('rdf:type owl:Class')
            
            # Count owl:ObjectProperty declarations
            stats['properties'] = owl_content.count('a owl:ObjectProperty') + owl_content.count('rdf:type owl:ObjectProperty')
            
            # Count owl:DatatypeProperty declarations
            stats['dataProperties'] = owl_content.count('a owl:DatatypeProperty') + owl_content.count('rdf:type owl:DatatypeProperty')
        except Exception:
            pass
        
        return stats
